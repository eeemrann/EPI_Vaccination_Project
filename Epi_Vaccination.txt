        <?php
        // Database credentials
        $host = 'localhost';
        $username = 'root';
        $password = '';
        $database = 'epi_vaccination';

        // Connect to the database
        $conn = new mysqli($host, $username, $password, $database);

        // Check connection
        if ($conn->connect_error) {
            die("<div class='alert alert-danger'>Connection failed: " . $conn->connect_error . "</div>");
        }

        // Handle form submission to insert data into the 'Child' table
        if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_POST['add_child'])) {
            $child_id = $_POST['child_id'];  // Child ID input field
            $child_name = $_POST['child_name'];
            $dob = $_POST['dob'];
            $address = $_POST['address'];
            $guardian_name = $_POST['guardian_name'];
            $contact_no = $_POST['contact_no'];
        
            $stmt = $conn->prepare("INSERT INTO Child (child_id, name, dob, age, address, guardian_name, contact_no) 
                                    VALUES (?, ?, ?, TIMESTAMPDIFF(YEAR, ?, CURDATE()), ?, ?, ?)");
            $stmt->bind_param("sssssss", $child_id, $child_name, $dob, $dob, $address, $guardian_name, $contact_no);
        
            if ($stmt->execute()) {
                echo "<div class='alert alert-success'>Child added successfully!</div>";
            } else {
                echo "<div class='alert alert-danger'>Error: " . $stmt->error . "</div>";
            }
        
            $stmt->close();
        }

        <?php
        // List of tables to show
        $tables = ['Child', 'Healthcare_Professional', 'Vaccination_Centre', 'Vaccination_Record', 'Vaccine'];

        foreach ($tables as $tableName) {
            echo "<h3 class='mt-5'>" . htmlspecialchars($tableName) . " Table</h3>";

            // Fetch data from the table
            $dataResult = $conn->query("SELECT * FROM $tableName");

            if (!$dataResult) {
                echo "<div class='alert alert-danger'>Error fetching data for table '$tableName': " . $conn->error . "</div>";
            } elseif ($dataResult->num_rows > 0) {
                echo "<table class='table table-bordered table-striped mt-4'>
                        <thead>
                            <tr>";
                // Display column headers
                $fields = $dataResult->fetch_fields();
                foreach ($fields as $field) {
                    echo "<th>" . htmlspecialchars($field->name) . "</th>";
                }
                echo "</tr>
                        </thead>
                        <tbody>" ;

                // Display table rows
                while ($row = $dataResult->fetch_assoc()) {
                    echo "<tr>";
                    foreach ($row as $cell) {
                        echo "<td>" . htmlspecialchars($cell) . "</td>";
                    }
                    echo "</tr>";
                }
                echo "</tbody>
                    </table>";
            } else {
                echo "<div class='alert alert-warning'>No data found in the table '$tableName'.</div>";
            }
        }

        // Close the connection
        $conn->close();
